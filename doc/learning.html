#' Creating model results
#'
#' @param data the lipidomics data set
#'
#' @return a data frame
calculate_estimates <- function(data) {
  data %>%
    split_by_metabolite() %>%
    map(generate_model_results) %>%
    list_rbind() %>%
    filter(str_detect(term, "metabolite_")) %>%
    add_original_metabolite_names(data)
}
